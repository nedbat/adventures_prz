    def salted_hash(salt, i):
        bytes = f"{salt}{i}".encode("ascii")
        return hashlib.md5(bytes).hexdigest()

Finding triples

    def triple(s):
        """Return the character that occurs as a triple, or None."""
        m = re.search(r"(.)\1\1", s)
        if m:
            return m.group(1)
